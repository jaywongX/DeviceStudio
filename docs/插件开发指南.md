# DeviceStudio 插件开发指南

> 版本：v1.0
> 状态：设计阶段

---

## 一、概述

DeviceStudio 采用插件架构，允许开发者扩展设备类型、协议解析、可视化组件和脚本命令。

### 1.1 插件类型

| 插件类型 | 接口 | 说明 |
|---------|------|------|
| 协议插件 | `IProtocolPlugin` | 扩展通信协议解析 |
| 设备插件 | `IDevicePlugin` | 扩展设备类型 |
| 可视化插件 | `IVisualizationPlugin` | 扩展数据可视化组件 |
| 脚本命令插件 | `IScriptCommandPlugin` | 扩展脚本命令 |

### 1.2 插件目录结构

```
plugins/
├── modbus/                    # Modbus 协议插件
│   ├── modbus.dll/.so
│   └── plugin.json
├── can/                       # CAN 总线插件
│   ├── can.dll/.so
│   └── plugin.json
└── custom_chart/              # 自定义图表插件
    ├── custom_chart.dll/.so
    └── plugin.json
```

### 3.2 CMakeLists.txt 模板

```cmake
cmake_minimum_required(VERSION 3.16)

project(MyProtocolPlugin VERSION 1.0.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)

# 查找 Qt
find_package(Qt6 REQUIRED COMPONENTS Core)

# 查找 DeviceStudio SDK
find_package(DeviceStudio REQUIRED)

# 插件源文件
set(SOURCES
    src/MyProtocolPlugin.cpp
    src/MyProtocol.cpp
)

set(HEADERS
    src/MyProtocolPlugin.h
    src/MyProtocol.h
)

# 创建动态库
add_library(${PROJECT_NAME} SHARED ${SOURCES} ${HEADERS})

# 链接库
target_link_libraries(${PROJECT_NAME} PRIVATE
    Qt6::Core
    DeviceStudio::Core
)

# 包含目录
target_include_directories(${PROJECT_NAME} PRIVATE
    ${CMAKE_CURRENT_SOURCE_DIR}/src
)

# 设置输出目录
set_target_properties(${PROJECT_NAME} PROPERTIES
    LIBRARY_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/plugins
    RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/plugins
)

# 安装规则
install(TARGETS ${PROJECT_NAME}
    LIBRARY DESTINATION plugins
    RUNTIME DESTINATION plugins
)
```

// MyProtocol.cpp

#include "MyProtocol.h"
#include <QJsonDocument>
#include <QJsonObject>
#include <QJsonArray>
#include <QFile>

    : IProtocol(parent)
{
}

bool MyProtocol::loadDefinition(const QString& filePath)
{
    QFile file(filePath);
    if (!file.open(QIODevice::ReadOnly)) {
        emit parseError(QString("无法打开文件: %1").arg(filePath), QByteArray());
        return false;
    }
    
    QByteArray data = file.readAll();
    file.close();
    
    return loadDefinitionFromJson(QString::fromUtf8(data));
}

bool MyProtocol::loadDefinitionFromJson(const QString& jsonString)
{
    QJsonParseError error;
    QJsonDocument doc = QJsonDocument::fromJson(jsonString.toUtf8(), &error);
    
    if (error.error != QJsonParseError::NoError) {
        emit parseError(QString("JSON解析错误: %1").arg(error.errorString()), QByteArray());
        return false;
    }
    
    // 解析帧定义
    QJsonObject root = doc.object();
    QJsonArray frames = root["frame_definitions"].toArray();
    
    m_frameDefinitions.clear();
    for (const auto& frameVal : frames) {
        FrameDefinition fd;
        QJsonObject frameObj = frameVal.toObject();
        
        fd.name = frameObj["name"].toString();
        fd.description = frameObj["description"].toString();
        fd.header = QByteArray::fromHex(frameObj["identification"].toObject()["header"].toString().toLatin1());
        fd.minFrameLength = frameObj["identification"].toObject()["min_length"].toInt(0);
        fd.maxFrameLength = frameObj["identification"].toObject()["max_length"].toInt(1024);
        
        // 解析字段...
        m_frameDefinitions.append(fd);
    }
    
    return true;
}

QList<ParsedFrame> MyProtocol::parse(const QByteArray& data, bool consume)
{
    QList<ParsedFrame> results;
    
    if (consume) {
        m_buffer.append(data);
    }
    
    // 尝试提取并解析完整帧
    while (true) {
        QByteArray frame = extractFrame();
        if (frame.isEmpty()) {
            break;
        }
        
        ParsedFrame parsed = parseFrame(frame);
        results.append(parsed);
        emit frameParsed(parsed);
    }
    
    return results;
}

ParsedFrame MyProtocol::parseFrame(const QByteArray& frameData)
{
    ParsedFrame result;
    result.rawData = frameData;
    result.timestamp = QDateTime::currentMSecsSinceEpoch();
    
    // 查找匹配的帧定义
    for (const auto& fd : m_frameDefinitions) {
        if (frameData.startsWith(fd.header) && 
            frameData.size() >= fd.minFrameLength &&
            frameData.size() <= fd.maxFrameLength) {
            
            result.frameType = fd.name;
            
            // 解析字段
            for (const auto& field : fd.fields) {
                if (field.offset + field.length <= frameData.size()) {
                    QByteArray fieldData = frameData.mid(field.offset, field.length);
                    
                    // 根据类型解析
                    QVariant value;
                    if (field.type == "uint8") {
                        value = static_cast<quint8>(fieldData[0]);
                    } else if (field.type == "uint16") {
                        value = qFromBigEndian<quint16>(fieldData.constData());
                    }
                    // ... 其他类型
                    
                    result.fields[field.name] = value;
                }
            }
            
            result.result = ParseResult::Success;
            return result;
        }
    }
    
    result.result = ParseResult::UnknownFrame;
    result.errorMessage = "未知的帧类型";
    return result;
}

QByteArray MyProtocol::extractFrame()
{
    // 查找帧头
    for (const auto& fd : m_frameDefinitions) {
        int startPos = m_buffer.indexOf(fd.header);
        if (startPos >= 0) {
            // 找到最小有效帧长度
            for (int len = fd.minFrameLength; len <= fd.maxFrameLength && startPos + len <= m_buffer.size(); ++len) {
                QByteArray potentialFrame = m_buffer.mid(startPos, len);
                if (validateFrame(potentialFrame)) {
                    m_buffer.remove(0, startPos + len);
                    return potentialFrame;
                }
            }
        }
    }
    
    return QByteArray();
}

#include "MyScriptCommands.h"

QList<ScriptCommand> MyScriptCommands::commands() const
{
    return {
        {
            "SENDFILE",
            "SENDFILE <device> <filepath>",
            "发送文件到设备",
            {"SENDFILE COM3 test.bin"},
            {"device", "filepath"}
        },
        {
            "WAITPATTERN",
            "WAITPATTERN <pattern> [timeout_ms]",
            "等待匹配的数据模式",
            {"WAITPATTERN \"OK\" 5000"},
            {"pattern", "timeout_ms"}
        },
        {
            "LOGDATA",
            "LOGDATA <message>",
            "记录数据到日志",
            {"LOGDATA \"测试完成\""},
            {"message"}
        }
    };
}

QVariant MyScriptCommands::executeCommand(const QString& command, 
                                          const QVariantList& args, 
                                          QVariantMap& context)
{
    if (command == "SENDFILE") {
        return cmdSendFile(args, context);
    } else if (command == "WAITPATTERN") {
        return cmdWaitForPattern(args, context);
    } else if (command == "LOGDATA") {
        return cmdLogData(args, context);
    }
    
    return QVariant();
}

QVariant MyScriptCommands::cmdSendFile(const QVariantList& args, QVariantMap& context)
{
    if (args.size() < 2) {
        return QVariant();
    }
    
    QString deviceId = args[0].toString();
    QString filePath = args[1].toString();
    
    // 获取设备
    IDevice* device = nullptr; // 从 context 获取
    if (!device) {
        return QVariant();
    }
    
    // 读取文件并发送
    QFile file(filePath);
    if (!file.open(QIODevice::ReadOnly)) {
        return QVariant();
    }
    
    QByteArray data = file.readAll();
    file.close();
    
    qint64 sent = device->write(data);
    return QVariant(sent);
}

## 十、最佳实践

### 10.1 性能优化

| 建议 | 说明 |
|-----|------|
| 避免频繁内存分配 | 预分配缓冲区 |
| 批量数据处理 | 累积后批量处理 |
| 异步操作 | 耗时操作放入线程 |
| 减少信号发射 | 合并高频更新 |

### 10.2 错误处理

```cpp
// 良好的错误处理示例
bool MyProtocol::loadDefinition(const QString& filePath)
{
    // 1. 参数校验
    if (filePath.isEmpty()) {
        qWarning() << "文件路径为空";
        return false;
    }
    
    // 2. 文件存在性检查
    if (!QFile::exists(filePath)) {
        emit parseError(QString("文件不存在: %1").arg(filePath), QByteArray());
        return false;
    }
    
    // 3. 尝试操作
    QFile file(filePath);
    if (!file.open(QIODevice::ReadOnly)) {
        emit parseError(QString("无法打开文件: %1 (%2)").arg(filePath).arg(file.errorString()), QByteArray());
        return false;
    }
    
    // 4. 资源清理 (RAII)
    // file 自动关闭
    
    return true;
}
```
