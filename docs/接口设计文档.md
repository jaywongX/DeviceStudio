## 一、文档概述

本文档定义 DeviceStudio 的核心模块接口和插件 API，为模块化开发和插件扩展提供规范。

### 1.1 设计原则

| 原则 | 说明 |
|-----|------|
| **接口隔离** | 每个接口职责单一，避免臃肿 |
| **依赖倒置** | 高层模块依赖抽象接口，不依赖具体实现 |
| **开闭原则** | 对扩展开放，对修改关闭 |
| **跨平台** | 接口设计考虑 Windows/Linux 兼容 |

### 1.2 命名规范

```cpp
// 接口类以 I 开头
class IDevice { ... };
class IProtocolPlugin { ... };

// 抽象基类以 Abstract 或 Base 开头
class AbstractDevice { ... };
class BaseProtocol { ... };

// 回调接口以 Callback 结尾
class IDataCallback { ... };
class IConnectionCallback { ... };

// 信号/槽使用 Qt 风格
signals:
    void dataReceived(const QByteArray& data);
    void connectionStatusChanged(bool connected);
```

## 二、核心模块接口

### 2.1 设备管理模块 (DeviceManager)

#### 2.1.1 IDevice 接口

设备抽象接口，所有设备类型（串口、TCP、UDP等）都实现此接口。

```cpp
// 文件: src/core/device/IDevice.h

#ifndef IDEVICE_H
#define IDEVICE_H

#include <QObject>
#include <QByteArray>
#include <QString>
#include <QVariantMap>

/**
 * @brief 设备类型枚举
 */
enum class DeviceType {
    Serial,     // 串口设备
    TcpClient,  // TCP 客户端
    TcpServer,  // TCP 服务器
    Udp,        // UDP
    Bluetooth,  // 蓝牙（扩展）
    Can,        // CAN 总线（扩展）
    Custom      // 自定义插件设备
};

/**
 * @brief 设备连接状态
 */
enum class DeviceState {
    Disconnected,   // 已断开
    Connecting,     // 连接中
    Connected,      // 已连接
    Disconnecting,  // 断开中
    Error           // 错误
};

/**
 * @brief 设备接口 - 所有设备类型的抽象基类
 */
class IDevice : public QObject
{
    Q_OBJECT

public:
    explicit IDevice(QObject* parent = nullptr) : QObject(parent) {}
    virtual ~IDevice() = default;

    // === 设备信息 ===
    
    /**
     * @brief 获取设备唯一标识符
     */
    virtual QString deviceId() const = 0;
    
    /**
     * @brief 获取设备名称
     */
    virtual QString name() const = 0;
    
    /**
     * @brief 设置设备名称
     */
    virtual void setName(const QString& name) = 0;
    
    /**
     * @brief 获取设备类型
     */
    virtual DeviceType type() const = 0;
    
    /**
     * @brief 获取设备状态
     */
    virtual DeviceState state() const = 0;
    
    /**
     * @brief 获取设备配置参数
     */
    virtual QVariantMap configuration() const = 0;
    
    /**
     * @brief 设置设备配置参数
     * @param config 配置参数映射
     * @return 是否设置成功
     */
    virtual bool setConfiguration(const QVariantMap& config) = 0;
    
    /**
     * @brief 获取设备描述信息（用于显示）
     */
    virtual QString description() const = 0;

    // === 连接管理 ===
    
    /**
     * @brief 打开设备连接
     * @return 是否成功发起连接
     */
    virtual bool open() = 0;
    
    /**
     * @brief 关闭设备连接
     */
    virtual void close() = 0;
    
    /**
     * @brief 检查设备是否已连接
     */
    virtual bool isOpen() const = 0;

    // === 数据收发 ===
    
    /**
     * @brief 发送数据
     * @param data 要发送的数据
     * @return 实际发送的字节数，-1 表示错误
     */
    virtual qint64 write(const QByteArray& data) = 0;
    
    /**
     * @brief 读取所有可用数据
     * @return 读取的数据
     */
    virtual QByteArray readAll() = 0;
    
    /**
     * @brief 读取指定字节数的数据
     * @param maxSize 最大读取字节数
     * @return 读取的数据
     */
    virtual QByteArray read(qint64 maxSize) = 0;
    
    /**
     * @brief 获取可读取的字节数
     */
    virtual qint64 bytesAvailable() const = 0;
    
    /**
     * @brief 清空接收缓冲区
     */
    virtual void clearReceiveBuffer() = 0;
    
    /**
     * @brief 清空发送缓冲区
     */
    virtual void clearSendBuffer() = 0;

    // === 统计信息 ===
    
    /**
     * @brief 获取已接收的字节总数
     */
    virtual qint64 totalBytesReceived() const = 0;
    
    /**
     * @brief 获取已发送的字节总数
     */
    virtual qint64 totalBytesSent() const = 0;
    
    /**
     * @brief 重置统计计数器
     */
    virtual void resetStatistics() = 0;

signals:
    /**
     * @brief 数据接收信号
     * @param data 接收到的数据
     */
    void dataReceived(const QByteArray& data);
    
    /**
     * @brief 数据发送完成信号
     * @param bytesSent 已发送字节数
     */
    void dataSent(qint64 bytesSent);
    
    /**
     * @brief 状态改变信号
     * @param state 新状态
     */
    void stateChanged(DeviceState state);
    
    /**
     * @brief 错误发生信号
     * @param errorString 错误描述
     */
    void errorOccurred(const QString& errorString);
};

// 元类型注册（用于 QVariant 和信号槽）
Q_DECLARE_METATYPE(DeviceType)
Q_DECLARE_METATYPE(DeviceState)

#endif // IDEVICE_H
```

### 2.2 协议引擎模块 (ProtocolEngine)

#### 2.2.1 IProtocol 接口

协议抽象接口，定义协议解析的基本行为。

```cpp
// 文件: src/core/protocol/IProtocol.h

#ifndef IPROTOCOL_H
#define IPROTOCOL_H

#include <QObject>
#include <QByteArray>
#include <QVariantMap>
#include <QString>

/**
 * @brief 解析结果状态
 */
enum class ParseResult {
    Success,        // 解析成功
    Incomplete,     // 数据不完整，需要更多数据
    InvalidHeader,  // 无效帧头
    InvalidCrc,     // CRC 校验失败
    InvalidLength,  // 长度错误
    UnknownFrame,   // 未知帧类型
    Error           // 其他错误
};

/**
 * @brief 协议字段定义
 */
struct ProtocolField {
    QString name;           // 字段名称
    QString type;           // 字段类型: uint8, int16, uint32, float, string, bytes
    int offset;             // 偏移量（字节）
    int length;             // 长度（字节）
    double scale = 1.0;     // 缩放因子
    double offsetValue = 0; // 偏移值
    QString encoding;       // 编码方式（用于字符串）
    QString description;    // 描述信息
    QVariant defaultValue;  // 默认值
    QVariant minValue;      // 最小值（用于数值范围检查）
    QVariant maxValue;      // 最大值（用于数值范围检查）
};

/**
 * @brief 帧定义
 */
struct FrameDefinition {
    QString name;                   // 帧名称
    QString description;            // 帧描述
    QByteArray header;              // 帧头
    int lengthFieldOffset = -1;     // 长度字段偏移（-1 表示固定长度）
    int lengthFieldSize = 1;        // 长度字段大小
    int minFrameLength = 0;         // 最小帧长度
    int maxFrameLength = 1024;      // 最大帧长度
    int payloadOffset = 0;          // 负载偏移
    int trailerLength = 0;          // 尾部长度
    QString crcAlgorithm;           // CRC 算法: none, crc8, crc16, crc32
    int crcOffset = -1;             // CRC 字段偏移
    QList<ProtocolField> fields;    // 字段列表
};

/**
 * @brief 解析后的数据帧
 */
struct ParsedFrame {
    QString frameType;              // 帧类型名称
    QByteArray rawData;             // 原始数据
    QVariantMap fields;             // 解析后的字段值
    ParseResult result;             // 解析结果
    QString errorMessage;           // 错误信息
    qint64 timestamp;               // 时间戳（毫秒）
};

/**
 * @brief 协议接口
 */
class IProtocol : public QObject
{
    Q_OBJECT

public:
    explicit IProtocol(QObject* parent = nullptr) : QObject(parent) {}
    virtual ~IProtocol() = default;

    // === 协议信息 ===
    
    virtual QString name() const = 0;
    virtual QString version() const = 0;
    virtual QString description() const = 0;

    // === 帧定义 ===
    
    /**
     * @brief 加载协议定义文件
     * @param filePath JSON 格式的协议定义文件路径
     * @return 是否加载成功
     */
    virtual bool loadDefinition(const QString& filePath) = 0;
    
    /**
     * @brief 加载协议定义（从 JSON 字符串）
     */
    virtual bool loadDefinitionFromJson(const QString& jsonString) = 0;
    
    /**
     * @brief 获取所有帧定义
     */
    virtual QList<FrameDefinition> frameDefinitions() const = 0;
    
    /**
     * @brief 根据名称获取帧定义
     */
    virtual FrameDefinition frameDefinition(const QString& name) const = 0;

    // === 数据解析 ===
    
    /**
     * @brief 解析数据
     * @param data 输入数据
     * @param consume 是否消费已解析的数据
     * @return 解析结果列表
     */
    virtual QList<ParsedFrame> parse(const QByteArray& data, bool consume = true) = 0;
    
    /**
     * @brief 解析单帧数据
     */
    virtual ParsedFrame parseFrame(const QByteArray& frameData) = 0;
    
    /**
     * @brief 尝试从缓冲区提取完整帧
     * @return 提取的完整帧，若无完整帧则返回空
     */
    virtual QByteArray extractFrame() = 0;
    
    /**
     * @brief 获取当前缓冲区数据
     */
    virtual QByteArray buffer() const = 0;
    
    /**
     * @brief 清空缓冲区
     */
    virtual void clearBuffer() = 0;

    // === 数据构建 ===
    
    /**
     * @brief 构建数据帧
     * @param frameType 帧类型
     * @param fields 字段值
     * @return 构建的帧数据
     */
    virtual QByteArray buildFrame(const QString& frameType, const QVariantMap& fields) = 0;

signals:
    /**
     * @brief 帧解析完成信号
     */
    void frameParsed(const ParsedFrame& frame);
    
    /**
     * @brief 解析错误信号
     */
    void parseError(const QString& error, const QByteArray& data);
};

Q_DECLARE_METATYPE(ParseResult)
Q_DECLARE_METATYPE(ProtocolField)
Q_DECLARE_METATYPE(FrameDefinition)
Q_DECLARE_METATYPE(ParsedFrame)

#endif // IPROTOCOL_H
```

```cpp
// 文件: src/core/script/IScriptEngine.h

#ifndef ISCRIPTENGINE_H

     * @brief 获取脚本内容

     * @brief 停止执行

     * @brief 获取当前执行状态

    virtual void bindDevice(const QString& varName, class IDevice* device) = 0;
    
    /**
     * @brief 解绑设备
     */
    virtual void unbindDevice(const QString& varName) = 0;

signals:
```

```cpp
#ifndef IDATACENTER_H
```

## 五、接口版本管理

### 5.1 版本兼容性

| 接口 | 当前版本 | 说明 |
|-----|---------|------|
| IDevice | 1.0 | 设备基础接口 |
| IDeviceManager | 1.0 | 设备管理接口 |
| IProtocol | 1.0 | 协议基础接口 |
| IProtocolEngine | 1.0 | 协议引擎接口 |
| IScriptEngine | 1.0 | 脚本引擎接口 |
| IDataCenter | 1.0 | 数据中心接口 |
| IPlugin | 1.0 | 插件基础接口 |

### 5.2 接口扩展原则

1. **新增接口**：继承现有接口，添加新方法
2. **修改接口**：通过默认参数或新方法实现向后兼容
3. **废弃接口**：标记为 `Q_DECL_DEPRECATED`，保留一个版本周期
4. **版本检查**：插件通过元数据声明支持的 API 版本

```cpp
// 示例：接口版本扩展
class IDeviceV2 : public IDevice
{
    Q_OBJECT
public:
    // 新增方法
    virtual QByteArray readLine() { return QByteArray(); }
    virtual bool waitForReadyRead(int msecs) { Q_UNUSED(msecs); return false; }
};
```

### 6.2 常用类型定义

```cpp
// 文件: src/utils/Types.h

#ifndef TYPES_H
#define TYPES_H

#include <QtGlobal>
#include <QString>

namespace DeviceStudio {

// 固定位宽类型
using Byte = quint8;
using Word = quint16;
using DWord = quint32;
using QWord = quint64;

// 字节数组工具函数
inline QString toHexString(const QByteArray& data, char separator = ' ') {
    QString result;
    for (int i = 0; i < data.size(); ++i) {
        if (i > 0 && separator) result.append(separator);
        result.append(QString("%1").arg(static_cast<quint8>(data[i]), 2, 16, QChar('0')).toUpper());
    }
    return result;
}

inline QByteArray fromHexString(const QString& hex) {
    QString clean = hex.simplified().remove(' ');
    return QByteArray::fromHex(clean.toLatin1());
}

} // namespace DeviceStudio

#endif // TYPES_H
```
